Write a c program of arthematic operators.

#include<stdio.h>
void main()
{
    float a,b,c;
    printf("Enter two numbers:");
    scanf("%f%f",&a,&b);
    c=a+b;
    printf("Sum of two numbers:%f\n",c);
    c=a-b;
    printf("Sub of two numbers:%f\n",c);
    c=a*b;
    printf("product of two numbers=%f\n",c);
    c=a/b;
    printf("Division of two numbers=%f\n",c);
}
